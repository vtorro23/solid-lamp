\chapter{Métodos de integración elemental para EDOS de primer orden}
Como se ha visto en el capítulo anterior, una EDO de primer orden viene dada generalmente por una expresión de la forma 
\[ F\left(t, x, x'\right)=0 .\]
En algunos casos podremos despejar $\displaystyle x' $ por lo que podremos escribir la EDO de la forma $\displaystyle x' = f\left(t,x\right) $. Esta última se denomina la \textbf{forma explícita o normal} de una EDO de primer orden. De esta forma podemos pasar fácilmente a la \textbf{forma diferencial} que en general tiene la forma
\[M\left(t,x\right)dt + N\left(t,x\right)dx = 0 .\]
Es fácil pasar de la forma explícita a la forma diferencial, en efecto, 
\[x' = f\left(t,x\right) \iff \frac{dx}{dt} = f\left(t,x\right) \iff f\left(t,x\right)dt-dx = 0 .\]
En general, veremos que dado el problema de Cauchy
\[ \left(P\right)
\begin{cases}
x' = f\left(t,x\right) \\
x\left(t_{0}\right)=x_{0}
\end{cases}
.\]
Tenemos entonces que resolver $\displaystyle P $ es equivalente a 
\[x\left(t\right)- x\left(t_{0}\right)= \int^{t}_{t_{0}} x'\left(s\right) \; d s = \int^{t}_{t_{0}} f\left(s, x\left(s\right)\right) \; d s  .\]

\section{Método de separación de variables}
\begin{definition}[EDO en variables separadas]
Se dice que una EDO está en \textbf{variables separadas} si podemos escribir 
\[x'\left(t\right) = f\left(t,x\left(t\right)\right) = p\left(t\right)q\left(x\left(t\right)\right) .\]
\end{definition}
\begin{observation}[Método de resolución]
	En general para resolver una EDO en variables separadas procedemos de la siguiente forma. Si $\displaystyle q\left(x\left(t\right)\right) \neq 0 $, podremos integrar en $\displaystyle t \in I $ \footnote{Donde $\displaystyle I $ es un intervalo donde buscamos que esté definida la solución.} a los dos lados de la igualdad. En efecto,
\[ x'\left(t\right) = p\left(t\right)q\left(x\left(t\right)\right) \iff \int \frac{x'\left(t\right)}{q\left(x\left(t\right)\right)} \; dt = \int p\left(t\right) \; dt \iff Q\left(x\left(t\right)\right) = P\left(t\right) + C,\]
donde $\displaystyle \frac{d}{dt}Q\left(x\left(t\right)\right) = \frac{x'\left(t\right)}{q\left(x\left(t\right)\right)} $ y $\displaystyle \frac{d}{dt}P\left(t\right) = p\left(t\right) $. Podemos aplicar la regla de la cadena para ver que 
\[\int \frac{x'\left(t\right)}{q\left(x\left(t\right)\right)} \; dt = \int \frac{1}{q\left(x\right)} \; dx .\]
\end{observation} 

\begin{theorem}[Existencia y unicidad para EDOS en variables separadas]
Si $\displaystyle f\left(t,x\right) = p\left(t\right)q\left(x\left(t\right)\right) $ con $\displaystyle p : \left(a,b\right)\to \R $ y $\displaystyle q : \left(c,d\right) \to \R $ continuas y $\displaystyle q\left(x\left(t\right)\right) \neq 0 $, $\displaystyle \forall x\left(t\right) \in \left(c,d\right) $, entonces 
\[ \left(P\right) 
\begin{cases}
x'\left(t\right) = p\left(t\right)q\left(x\left(t\right)\right) \\
x\left(t_{0}\right) = x_{0}
\end{cases}
\]
tiene solución única para todo $\displaystyle \left(t_{0}, x_{0}\right) \in \left(a,b\right)\times \left(c,d\right) $, en un entorno de $\displaystyle t_{0} $.
\end{theorem}

\begin{proof}
	 Sea $\displaystyle x\left(t\right) $ una solución del problema. Como $\displaystyle q\left(x\right) \neq 0$ en $\displaystyle \left(c,d\right) $ podemos escribir
	\[\frac{x'\left(t\right)}{q\left(x\left(t\right)\right)} = p\left(t\right) .\]
Sean 
\[Q\left(x\right) := \int ^{x}_{x_{0}}\frac{1}{q\left(s\right)} \; d s \quad \text{y} \quad P\left(t\right):= \int ^{t}_{t_{0}}p\left(s\right) \; d s .\]
Por el teorema fundamental del cálculo tendremos que $\displaystyle \frac{d}{dt}Q\left(x\left(t\right)\right) = \frac{x'}{q\left(x\left(t\right)\right)} $, por lo que la EDO equivale a $\displaystyle \frac{d}{dt}Q\left(x\left(t\right)\right) = \frac{d}{dt}P\left(t\right) $. Integrando obtenemos 
\[Q\left(x\left(t\right)\right)= P\left(t\right) + C .\]
Imponiendo la condición inicial $\displaystyle x\left(t_{0}\right)=x_{0} $ tenemos que $\displaystyle Q\left(x_{0}\right) = P\left(t_{0}\right) = 0 $, por lo que $\displaystyle C = 0 $ y nos queda que $\displaystyle Q\left(x\left(t\right)\right)=P\left(t\right) $. Ahora, definimos
\[F\left(t,x\right)= Q\left(x\right)-P\left(t\right) .\]
Se cumple:
\begin{itemize}
\item $\displaystyle F\left(t_{0}, x_{0}\right)=0 $.
\item $\displaystyle F \in \mathcal{C}^{1}\left(\left(a, b\right)\times \left(c,d\right)\right) $.
\item $\displaystyle \frac{\partial F}{\partial x}\left(t_{0}, x_{0}\right)= \frac{1}{q\left(x_{0}\right)} \neq 0 $.
\end{itemize}
Por el Teorema de la Función Implícita existe un entorno de $\displaystyle t_{0} $ en el que existe una única solución $\displaystyle x\left(t\right) $ tal que $\displaystyle F\left(t,x\left(t\right)\right)=0 $. Si derivamos implícitamente obtenemos que 
\[F_{t} + x'\left(t\right)F_{x}= 0 \Rightarrow x'\left(t\right)= - \frac{F_{t}}{F_{x}} = p\left(t\right)q\left(x\left(t\right)\right) .\]
La unicidad se deduce del Teorema de la Función Implícita. 
\end{proof}
\begin{eg}
Obtengamos la solución general de la EDO $\displaystyle x' = \frac{x^{2}}{t} $ y la solución particular cuando $\displaystyle x\left(1\right) = 0 $ . Como $\displaystyle x\left(1\right)=0 $, no podemos aplicar el método de separación de variables, por lo que la solución particular que buscamos es la solución trivial. Para resolver la EDO (sin condición inicial), una vez descartada la solución trivial, tendremos que
\[ x' = \frac{x^{2}}{t} \iff \int \frac{x'}{x^{2}} \; dt = \int \frac{1}{t} \; dt \iff -\frac{1}{x} = \ln t + C \iff x = \frac{1}{C - \ln t}.\]
\end{eg}
\begin{eg}
Obtengamos la solución del problema
\[\left(P\right)
\begin{cases}
x' = \frac{\left(2t+1\right)\left(2x-1\right)}{2\left(t ^{2} + t\right)} \\
x\left(1\right) = 0
\end{cases}
.\]
Podemos observar que es de variables separadas, además la solución $\displaystyle x \equiv \frac{1}{2} $ es solución de la EDO pero no del problema. Tendremos que para $\displaystyle t \in I $ donde $\displaystyle I $ es un entorno del 1:
\[
\begin{split}
	\frac{2x'}{2x-1} = \frac{2t+1}{t ^{2} + t} \iff & \int \frac{2x'}{2x-1} \; dt = \int \frac{2t+1}{t ^{2}+t} \; dt \iff \ln (1-2x)= \ln (t ^{2}+t)+ C \\
	\iff & \ln \left(\frac{1-2x}{t ^{2}+t}\right) = C \iff  \frac{1-2x}{t^{2}+t}=C.
\end{split}
\]
En un entorno de $\displaystyle \left(1,0\right) $ tenemos que los contenidos del valor absoluto son positivos por lo que podemos decir que $\displaystyle C = \frac{1}{2} $. 
\end{eg}
\section{EDOS lineales}

Sea $\displaystyle \left(E\right) \; x'\left(t\right) = a\left(t\right)x\left(t\right) + b\left(t\right) $ una EDO lineal. Tendremos que la EDO lineal homogénea asociada será $\displaystyle \left(E_{h}\right) \; x'\left(t\right) = a\left(t\right)x\left(t\right) $. Consideraremos que $\displaystyle a\left(t\right) $ y $\displaystyle b\left(t\right) $ son continuas en un cierto intervalo. Claramente esta última es una EDO de variables separadas. Tendremos que, suponiendo que $\displaystyle x\left(t\right)\neq 0 $,
\[ \int \frac{x'\left(t\right)}{x\left(t\right)} \; dt = \int a\left(t\right) \; dt \iff \ln x\left(t\right) = \int a\left(t\right) \; dt + C \iff x\left(t\right) = C\exp\left(\int a\left(t\right) \; dt\right) .\]
Estas son todas las soluciones de la EDO. En efecto, podemos ver que todas las soluciones son proporcionales a la función $\displaystyle \exp\left(\int a\left(t\right) \; dt\right) $. Si $\displaystyle u\left(t\right) $ es solución de $\displaystyle \left(E_{h}\right) $ podemos ver que si cogemos
\[u\left(t\right)\exp\left(-\int a\left(s\right) \; d s\right) \]
y lo derivamos, tendremos que 
\[u'\left(t\right)\exp\left(-\int a\left(s\right) \; d s\right) - u\left(t\right)a\left(t\right)\exp\left(-\int a\left(s\right) \; d s\right) .\]
Como $\displaystyle u\left(t\right) $ es solución tendremos que $\displaystyle u'\left(t\right)= a\left(t\right)u\left(t\right) $, por lo que 
\[ = u\left(t\right)a\left(t\right)\exp\left(-\int a\left(s\right) \; d s\right) - u\left(t\right)a\left(t\right)\exp\left(-\int a\left(s\right) \; d s\right) = 0 .\]
Por tanto, tendremos que 
\[ u\left(t\right)\exp\left(-\int a\left(s\right) \; d s\right)= K \Rightarrow u\left(t\right) = K\exp\left(\int a\left(s\right) \; d s\right) .\]
Así, hemos demostrado que esta familia monoparamétrica compone todas las soluciones de la EDO. Es fácil comprobar que las soluciones de $\displaystyle \left(E_{h}\right) $ constituyen un $\displaystyle \R $-espacio vectorial de dimensión 1.
\begin{prop}
El conjunto de soluciones de $\displaystyle \left(E_{h}\right) $ tiene estructura de $\displaystyle \R $-espacio vectorial de dimensión 1.
\end{prop}
\begin{proof}
	En el capítulo anterior vimos que el conjunto de soluciones de una EDO lineal componía un $\displaystyle \R $-espacio vectorial, falta por ver que en este caso la dimensión será 1 (el argumento que hemos dado anteriormente vale). \\
Supongamos que $\displaystyle u\left(t\right)=c\left(t\right)\exp\left(\int a\left(s\right) \; d s\right) $ es solución de $\displaystyle \left(E_{h}\right) $. Por tanto, debe ser que $\displaystyle u'\left(t\right)=a\left(t\right)u\left(t\right) $, por lo que 
\[c'\left(t\right)\exp\left(\int a\left(s\right) \; d s\right) + c\left(t\right)a\left(t\right)\exp\left(\int a\left(s\right) \; d s\right) = a\left(t\right)c\left(t\right)\exp\left(\int a\left(s\right) \; d s\right) .\]
Por tanto, debe ser que $\displaystyle c'\equiv 0 $, por lo que $\displaystyle c\left(t\right) $ es una función constante. 
\end{proof}
\begin{colorary}
La solución general de $\displaystyle \left(E\right) $ es de la forma 
\[x\left(t\right)=x_{h}\left(t\right)+x_{p}\left(t\right) ,\]
donde $\displaystyle x_{h}\left(t\right) $ es la solución general de $\displaystyle \left(E_{h}\right) $ y $\displaystyle x_{p}\left(t\right) $ es una solución particular de $\displaystyle \left(E\right) $.
\end{colorary}
\begin{proof}
Si $\displaystyle x_{h} $ es solución de $\displaystyle \left(E_{h}\right) $ tendremos que $\displaystyle x_{h}' = a\left(t\right)ax_{h} $. Además, si $\displaystyle x_{p}\left(t\right) $ es solución particular de $\displaystyle \left(E\right) $ tendremos que $\displaystyle x_{p}' = a\left(t\right)x_{p} + b\left(t\right) $. Así, tenemos que
\[\left(x_{h} + x_{p}\right)' = x_{h}' + x_{p} ' = a\left(t\right)\left(x_{h} + x_{p}\right) + b\left(t\right).\]
Por tanto, $\displaystyle x_{h}+x_{p} $ es solución de $\displaystyle \left(E\right) $. Veamos ahora que cualquier solución de $\displaystyle \left(E\right) $ se puede expresar de esta forma. Si $\displaystyle x_{p}\left(t\right) $ es solución particular de $\displaystyle \left(E\right) $ tendremos que 
\[\left(x\left(t\right) - x_{p}\left(t\right)\right)' = x' - x_{p}' = a\left(t\right)x + b\left(t\right)-a\left(t\right)x_{p} -b\left(t\right) = a\left(t\right)\left(x-x_{p}\right) .\]
Por tanto, tenemos que la diferencia es solución de $\displaystyle \left(E_{h}\right) $. 
\end{proof}

\begin{observation}
En este caso, el conjunto de soluciones forma un espacio afín.
\end{observation}

\begin{observation}[Método de variación de constantes]
Para resolver la EDO completa nos falta ver cómo obtener una solución particular. Esto lo haremos mediante el \textbf{método de variación de constantes:} proponemos una solución particular no muy diferente a $\displaystyle x_{h}\left(t\right) $, como por ejemplo
\[x_{p}\left(t\right) = C\left(t\right)\exp\left(\int a\left(s\right) \; d s\right) .\]
Forzamos que sea solución y tendremos que
\[x'_{p} = C'\exp\left(\int a\left(s\right) \; d s\right)+ Ca\left(t\right)\exp\left(\int a\left(s\right) \; d s\right) .\]
Queremos que $\displaystyle x'_{p}\left(t\right) = a\left(t\right)x_{p}+b\left(t\right) $, pero esto tiene como consecuencia que 
\[C'\exp\left(\int a\left(s\right) \; d s\right) + Ca\left(t\right)\exp\left(\int a\left(s\right) \; d s\right) = a\left(t\right)C\exp\left(\int a\left(s\right) \; d s\right) +b\left(t\right).\]
De esta forma, nos queda que 
\[C'\left(t\right) = b\left(t\right)\exp\left(-\int a\left(s\right) \; d s\right) \iff C\left(t\right)=\int b\left(t\right)\exp\left(-\int a\left(s\right) \; d s\right) \; dt .\]
\end{observation}
Así, hemos demostrado el siguiente teorema:
\begin{theorem}
Para cada $\displaystyle \left(t_{0}, x_{0}\right) \in \left(\alpha, \omega \right)\times \R $ \footnote{Siendo $\displaystyle \left(\alpha, \omega \right) $ el intervalo donde son continuas $\displaystyle a\left(t\right) $ y $\displaystyle b\left(t\right) $.} el problema
\[
\begin{cases}
x'\left(t\right)=a\left(t\right)x \\
x\left(t_{0}\right)=x_{0}
\end{cases}
\]
admite por solución $\displaystyle x\left(t\right)=x_{0}\exp\left(\int^{t}_{t_{0}} a\left(s\right) \; d s \right) $. Análogamente, la única solución del problema
\[
\begin{cases}
x'\left(t\right)= a\left(t\right)x + b\left(t\right) \\
x\left(t_{0}\right)=x_{0}
\end{cases}
\]
es la función
\[x\left(t\right)=C\exp\left(\int^{t}_{t_{0}} a\left(s\right) \; d s\right) + \exp\left(\int^{t}_{t_{0}} a\left(s\right) \; d s\right)\int^{t}_{t _{0}} b\left(s\right)\exp\left(-\int^{s}_{t_{0}} a\left(u\right) \; du\right) \; d s .\]
\end{theorem}

\begin{eg}[Hoja2, Ejercicio 6]
Consideremos la EDO
\[\left(R\right) \quad x' + x = 2te^{-t} + t ^{2} .\]
En este caso, tendremos que $\displaystyle \left(E\right) $ es $\displaystyle x' +x = 2te^{-t}+t ^{2} $ y $\displaystyle \left(E_{h}\right) $ será $\displaystyle x' + x = 0 $. Resolvemos $\displaystyle \left(E_{h}\right) $ por separación de variables:
\[\frac{x'}{x} = -1 \iff \int \frac{x'}{x} \; dt = -\int  \; dt \iff \ln x = -t + C \iff x_{h}\left(t\right) = Ce^{-t}, C \in \R .\]
Podemos observar que la solución trivial está incluída dentro de nuestra familia monoparamétrica de soluciones. Ahora, buscamos una solución particular $\displaystyle x_{p}\left(t\right) $ por el método de variación de constantes. Es decir, buscamos $\displaystyle x_{p}\left(t\right)=C\left(t\right)e^{-t} $. Tendremos que 
\[x_{p}'\left(t\right)= C'\left(t\right)e^{-t}-C\left(t\right)e^{-t}  .\]
Queremos que $\displaystyle x'_{p} + x = 2te^{-t}+t ^{2} $, por lo que 
\[C'\left(t\right)e^{-t}-Ce^{-t} + Ce^{-t} = 2te^{-t} + t ^{2} \iff C'\left(t\right) = 2t + t ^{2}e^{t}.\]
Así, nos queda que 
\[C\left(t\right) = \int 2t + t ^{2}e^{t} \; dt = t ^{2} + t ^{2}e^{t}-2te^{t}+2e^{t} .\]
Por tanto, nos queda que 
\[x_{p}\left(t\right)=t ^{2}e^{-t} + t ^{2}-2t+2 .\]
Así, la solución general de $\displaystyle \left(E\right) $ será
\[x\left(t\right)=Ce^{-t}+t ^{2}e^{-t} + t ^{2}-2t + 2 .\]
Calculemos ahora la solución que verifique $\displaystyle x\left(0\right) = 1 $. Tendremos que $\displaystyle C = -1 $, por lo que la solución particular que buscamos será
\[x\left(t\right)=e^{-t} + t ^{2}e^{-t}+t ^{2}-2t + 2 .\]
\end{eg}
\section{Ecuaciones homogéneas}
\begin{definition}[Función homogénea]
Decimos que $\displaystyle f\left(t,x\right) $ es una \textbf{función homogénea} de grado $\displaystyle m \in \R $ si \footnote{Esta definición se puede extender a funciones de más variables.} 
\[f\left(\lambda t, \lambda x\right)= \lambda^{m}f\left(t,x\right), \; \forall \lambda \in \R .\]
\end{definition}
\begin{eg}
La función $\displaystyle f\left(t,x\right) = t ^{2}+x^{2} $ es homogénea de grado 2, sin embargo $\displaystyle f\left(t,x\right) = t ^{2} + x^{2} + 3 $ no es homogénea.
\end{eg}
\begin{definition}[EDO homogénea]
Dada la EDO $\displaystyle x'=f\left(t,x\right) $, decimos que es una \textbf{EDO homogénea} si $\displaystyle f $ es una función homogénea de grado 0.	
\end{definition}
\begin{observation}
Podemos ver que esto es equivalente a que, si la EDO viene dada por $\displaystyle M\left(t,x\right)dt + N\left(t,x\right)dx = 0 $ en su forma diferencial, entonces $\displaystyle M $ y $\displaystyle N $ son funciones homogéneas del mismo grado. En efecto, si $\displaystyle N\left(t,x\right)\neq 0 $ tendremos que 
\[ x'= -\frac{M\left(\lambda t,\lambda x\right)}{N\left(\lambda t,\lambda x\right)} = -\frac{\lambda^{m}M\left(t,x\right)}{\lambda^{m}N\left(t,x\right)}=-\frac{M\left(t,x\right)}{N\left(t,x\right)}.\]
\end{observation}
\begin{observation}[Método de resolución]
Veremos que las EDOS homogéneas son EDOS de variables separadas haciendo el cambio de variable oportuno. En efecto, podemos ver que 
\[x' = f\left(t,x\right)=f\left(t,t\frac{x}{t}\right) = t ^{0}f\left(1,\frac{x}{t}\right) = f\left(1, \frac{x}{t}\right).\]
Haciendo el cambio de variable $\displaystyle z\left(t\right) = \frac{x\left(t\right)}{t} $, tenemos que 
\[x=zt \Rightarrow x' = z't + z .\]
Por tanto, tendremos que $\displaystyle z't + z = f\left(1,z\right) = \tilde{f}\left(z\right) $, que es una EDO de variables separadas, por lo que podemos proceder de la forma
\[z't = \tilde{f}\left(z\right) -z \iff \frac{z'}{\tilde{f}\left(z\right)-z}=\frac{1}{t} .\]
\end{observation}

\begin{eg}[Hoja2, Ejercicio 3 apartado 1]
Consideremos el problema de valor inicial
\[
\begin{cases}
t x^{2}x' = x^{3}-t ^{3} \\
x\left(1\right) = 2
\end{cases}
.\]
Tenemos que la EDO es equivalente a $\displaystyle tx^{2}dx +\left(t ^{3}-x^{3}\right)dt = 0 $ y ambas $\displaystyle M $ y $\displaystyle N $ son homogéneas de grado 3. Tenemos que 
\[x'=\frac{x^{3}-t ^{3}}{tx^{2}} = \frac{\left(\frac{x}{t}\right)^{3}-1}{\left(\frac{x}{t}\right)^{2}} .\]
Hacemos el cambio de variable $\displaystyle z = \frac{x}{t} $, de forma que $\displaystyle x' = z't + z $. Así, nos queda que para $\displaystyle t > 0 $,
\[z't + z = \frac{z^{3}-1}{z^{2}} \iff z't = -\frac{1}{z^{2}} \iff z^{2}z' = -\frac{1}{t} \iff \frac{z^{3}}{3}= - \ln t + C, \; C \in \R .\]
Deshaciendo el cambio $\displaystyle z = \frac{x}{t} $, tenemos que 
\[x^{3} = -3t ^{3}\ln t + Ct ^{3}, \; C \in \R, \; t > 0 .\]
Ahora buscamos la solución particular
\[8 = -3 \cdot 1 \cdot \ln 1 + C \cdot 1 \iff C = 8 .\]
Así, nos queda que la solución particular será $\displaystyle x^{3}= -3t ^{3}\ln t + 8 t ^{3} $ con $\displaystyle t > 0 $. 
\end{eg}

\section{Ecuaciones exactas y factor integrante}
\begin{definition}[Ecuación exacta]
La EDO $\displaystyle M\left(t,x\right)dt + N\left(t,x\right)dx = 0 $ con $\displaystyle M $ y $\displaystyle N $ continuas en $\displaystyle \Omega\subset\R^{2} $, se dice que es \textbf{exacta} si existe si existe una función $\displaystyle F : \Omega \subset \R^{2} \to \R $ con $\displaystyle F \in \mathcal{C}^{1}\left(\Omega \right) $ tal que la solución de la EDO son las curvas de nivel de $\displaystyle F $, esto es $\displaystyle F\left(t,x\right)=C $, $\displaystyle C \in \R $.
\end{definition}
\begin{observation}
La definición equivalente a la anterior es: \\

\textit{La EDO $\displaystyle M\left(t,x\right)dt + N\left(t,x\right)dx = 0 $ es \textbf{exacta} si existe $\displaystyle F : \Omega \subset \R^{2} \to \R $ con $\displaystyle F \in \mathcal{C}^{1}\left(\Omega \right) $ tal que}
\[\frac{\partial F\left(t,x\right)}{\partial t}= M\left(t,x\right) \quad \text{y} \quad \frac{\partial F\left(t,x\right)}{\partial x}=N\left(t,x\right) .\]
En efecto, derivando implícitamente tenemos que 
\[\frac{\partial F}{\partial t} + \frac{\partial F}{\partial x}x'= 0 \iff \frac{\partial F}{\partial t}dt + \frac{\partial F}{\partial x}dx = 0.\]
Así, tendremos que $\displaystyle M\left(t,x\right)=\frac{\partial F}{\partial t} $ y $\displaystyle N\left(t,x\right)= \frac{\partial F}{\partial x} $. Recíprocamente, 
\[M\left(t,x\right) + N\left(x,t\right)\frac{dx}{dt} = \frac{\partial F\left(t,x\right)}{\partial t} + \frac{\partial F\left(t,x\right)}{\partial x}\frac{dx}{dt}=\frac{d}{dt}F\left(t,x\right) ,\]
por lo que la EDO es equivalente a $\displaystyle \frac{d}{dt}F\left(t,x\right)=0 $ y sus soluciones son las curvas de nivel del campo $\displaystyle F\left(t,x\right) $. 
\end{observation}
\begin{prop}
Para $\displaystyle M $ y $\displaystyle N $ continuas y con derivadas parciales continuas en $\displaystyle \Omega\subset \R^{2} $, la EDO $\displaystyle M\left(t,x\right)dt + N\left(t,x\right)dx = 0 $ es exacta si y solo si $\displaystyle \frac{\partial M}{\partial x} = \frac{\partial N}{\partial t} $. 
\end{prop}
\begin{proof}
	Como $\displaystyle M,N \in \mathcal{C}^{1} $, se sigue que $\displaystyle F\in\mathcal{C}^{2}\left(\Omega \right) $. 
\begin{description}
\item[(i)] Si $\displaystyle M\left(t,x\right)dt + N\left(t,x\right)dx = 0 $ es exacta, existe $\displaystyle F : \Omega\subset\R^{2} \to \R$ tal que 
	\[M\left(t,x\right)=\frac{\partial F}{\partial t}\quad \text{y} \quad N\left(t,x\right)=\frac{\partial F}{\partial x} .\]
Como $\displaystyle F \in \mathcal{C}^{2}\left(\Omega \right) $ podemos aplicar el teorema de Schwartz, por lo que 
\[\frac{\partial M}{\partial x}= \frac{\partial }{\partial x}\left(\frac{\partial F}{\partial t}\right)= \frac{\partial }{\partial t}\left(\frac{\partial F}{\partial x}\right)= \frac{\partial N}{\partial t} .\]
\item[(ii)] Supongamos que $\displaystyle \frac{\partial M}{\partial x}= \frac{\partial N}{\partial t} $. Buscamos $\displaystyle F : \Omega\subset \R^{2} \to \R $ tal que $\displaystyle F \in \mathcal{C}^{2}\left(\Omega \right) $ y $\displaystyle \frac{\partial F}{\partial t}=M\left(t,x\right) $ y $\displaystyle \frac{\partial F}{\partial x}= N\left(t,x\right) $. Podemos tomar
	\[\frac{\partial F}{\partial t}=M\left(t,x\right) \Rightarrow F\left(t,x\right)=\int M\left(t,x\right) \; dt + C\left(x\right) .\]
Tendremos que $\displaystyle C\left(x\right) $ es tal que 
\[\frac{\partial F}{\partial x}= \frac{\partial }{\partial x}\left(\int M\left(t,x\right) \; dt\right)+C'\left(x\right)=N\left(t,x\right) \iff C'\left(x\right)=N\left(t,x\right)-\int \frac{\partial }{\partial x}M\left(t,x\right) \; dt .\]
De esta forma tendremos que 
\[C\left(x\right)=\int \left(M\left(t,x\right)-\int \frac{\partial }{\partial x}N\left(t, x\right) \; dt\right) \; dx + C .\]
La construcción está bien hecha y $\displaystyle C\left(x\right) $ es una función que depende sólo de $\displaystyle x $ puesto que si derivamos $\displaystyle C'\left(x\right) $ respecto de $\displaystyle t $ obtenemos 
\[\frac{\partial C'}{\partial t} = \frac{\partial M}{\partial t}-\frac{\partial N}{\partial x}= 0 .\]
Así, tenemos que 
\[F\left(t,x\right)=\int M\left(t,x\right) \; dt + \int \left(N\left(t,x\right) -\int \frac{\partial }{\partial x}M\left(t,x\right) \; dt\right) \; dx + C .\]
\end{description}
\end{proof}
\begin{observation}
Observando la construcción que acabamos de hacer tendremos que la solución general de la ecuación es $\displaystyle F\left(t,x\right)=C $, donde $\displaystyle F $ la hemos calculado anteriormente. 
\end{observation}

\begin{eg}[Hoja 2, Ejercicio 4 apartado (a)]
Consideremos $\displaystyle x' = \frac{2tx}{x - t ^{2}} $. En su forma diferencial tendremos que
\[2tx \; dt +\left(t ^{2}-x\right) \; dx = 0 .\]
Veamos si es exacta. Tenemos que $\displaystyle M\left(t,x\right)=2tx $ y $\displaystyle N\left(t,x\right)=t ^{2}-x $, así
\[\frac{\partial M}{\partial x}=2t = \frac{\partial N}{\partial t} .\]
Como es exacta, existe $\displaystyle F : \Omega\subset \R^{2} \to \R $ tal que la solución de la EDO son las curvas de nivel de $\displaystyle F $. Buscamos $\displaystyle F $:
\[\frac{\partial F}{\partial t}= 2tx \quad \text{y} \quad \frac{\partial F}{\partial x} = t ^{2}-x .\]
Tendremos que 
\[ F\left(t,x\right)=\int 2tx \; dt + C\left(x\right)= t ^{2}x + C\left(x\right) .\]
Además, necesitamos que 
\[\frac{\partial F}{\partial x}=t ^{2}-x=t ^{2}+ C'\left(x\right) \iff C'\left(x\right)=-x .\]
Así, tenemos que $\displaystyle C\left(x\right)=-\frac{x^{2}}{2} $ \footnote{No nos interesan las constantes porque las soluciones son las curvas de nivel de $\displaystyle F $.} , por lo que nos queda que $\displaystyle F\left(t,x\right)=t ^{2}x - \frac{x^{2}}{2} $ y la solución de la EDO es
\[ t ^{2}x - \frac{x^{2}}{2} = C, \; C \in \R.\]
\end{eg}
\begin{observation}[Factor integrante]
Hay EDOS que no son exactas, pero que si las multiplicamos por una función $\displaystyle \mu\left(t,x\right) $ se convierten en exactas. Es decir, $\displaystyle M\left(t,x\right)dt + N\left(t,x\right)dx=0 $ no es exacta, pero existe $\displaystyle \mu $ tal que 
\[\underbrace{\mu M\left(t,x\right)}_{\widetilde{M}} + \underbrace{\mu N\left(t,x\right)}_{\tilde{N}} x' = 0 ,\]
es exacta puesto que $\displaystyle \frac{\partial \widetilde{M}}{\partial x}= \frac{\partial \tilde{N}}{\partial t} $. A $\displaystyle \mu $ lo llamamos \textbf{factor integrante}. A la hora de buscar uno, buscamos que se cumpla la condición
\[\frac{\partial M}{\partial x}\mu + M\frac{\partial \mu}{\partial x}= \frac{\partial N}{\partial t}\mu + N\frac{\partial \mu}{\partial t} .\]
Si no nos dan alguna información de $\displaystyle \mu $ para que lo podeamos calcular, no podemos calcular $\displaystyle \mu $ ni la solución de la EDO (sería una EDP muy complicada).
\end{observation}
\begin{eg}[Hoja 2, Ejercicio 10]
Consideremos la EDO 
\[\left(x\cos \frac{x}{t}-t\right)dt - t \cos \frac{x}{t}dx = 0 .\]
Veamos si es exacta y si no lo es buscaremos un factor integrante $\displaystyle \mu\left(t,x\right)=\mu\left(\frac{x}{t}\right) $. Tenemos que $\displaystyle M\left(t,x\right)=x\cos\frac{x}{t} - t $ y $\displaystyle N\left(t,x\right)=-t \cos \frac{x}{t} $. Tenemos que 
\[\frac{\partial M}{\partial x} = \cos \frac{x}{t} + -\frac{x}{t}\sin \frac{x}{t}, \quad \frac{\partial N}{\partial t}=-\cos \frac{x}{t} - \frac{x}{t}\sin \frac{x}{t} .\]
Como $\displaystyle \frac{\partial M}{\partial x} \neq \frac{\partial N}{\partial t} $, tenemos que la EDO no es exacta. Buscamos el factor integrante. Tendremos que $\displaystyle \widetilde{M}\left(t,x\right)=\mu\left(\frac{x}{t}\right)\left(x\cos \frac{x}{t}-t\right) $ y $\displaystyle \tilde{N}\left(t,x\right)=-\mu\left(\frac{x}{t}\right)t \cos \frac{x}{t} $. Para que sea exacta necesitamos que $\displaystyle \frac{\partial \widetilde{M}}{\partial x} = \frac{\partial \tilde{N}}{\partial t} $. 
\[ \frac{\partial \widetilde{M}}{\partial x} = \frac{1}{t}\mu'\left(\frac{x}{t}\right)\left(x\cos \frac{x}{t}-t\right)+\mu\left(\frac{x}{t}\right)\left(\cos \frac{x}{t}- \frac{x}{t}\sin \frac{x}{t}\right) .\]
\[\frac{\partial \tilde{N}}{\partial t}= \frac{x}{t ^{2}}\mu'\left(\frac{x}{t}\right)t\cos \frac{x}{t} - \mu\left(\frac{x}{t}\right)\left(\cos \frac{x}{t} + \frac{x}{t} \sin \frac{x}{t}\right) .\]
Para que coincidan ambos valores tenemos que 
\[ 2\mu\left(\frac{x}{t}\right)\cos\frac{x}{t} = \mu'\left(\frac{x}{t}\right).\]
Así, nos queda una EDO de separación de variables. Cogemos $\displaystyle s = \frac{x}{t} $ y tenemos que 
\[\frac{\mu'\left(s\right)}{\mu\left(s\right)} = 2\cos s \iff \int \frac{1}{\mu} \; d\mu = \int 2\cos s \; d s \iff \ln \mu = 2\sin s \iff \mu\left(\frac{x}{t}\right) = Ce^{2\sin \frac{x}{t}} .\]
Podemos coger cualquier factor integrante, por lo que podemos coger $\displaystyle \mu \left(\frac{x}{t}\right)=\exp\left(2\sin \frac{x}{t}\right) $. Ahora podemos resolver la EDO $\displaystyle \widetilde{M}\left(t,x\right)dt + \tilde{N}\left(t,x\right)dx = 0 $ como una EDO exacta. Tenemos que existe $\displaystyle F : \Omega \subset \R^{2} \to \R $ tal que 
\[\frac{\partial F}{\partial t}=\tilde{M}\left(t,x\right)=\exp\left(2\sin \frac{x}{t}\right)\left(x\cos \frac{x}{t}-t\right) \quad \text{y} \quad \frac{\partial F}{\partial x}= \tilde{N}\left(t,x\right) = -\exp\left(2\sin \frac{x}{t}\right)t \cos \frac{x}{t}.\]
Tendremos que 
\[F\left(t,x\right) = -t\int \exp\left(2\sin \frac{x}{t}\right)\cos \frac{x}{t} \; dx = - \frac{t ^{2}}{2}\exp\left(2\sin \frac{x}{t}\right) + C\left(t\right).\]
Para calcular $\displaystyle C\left(t\right) $ tenemos que 
\[\frac{\partial F}{\partial t}=-t\exp\left(2\sin \frac{x}{t}\right) + x\exp\left(\sin \frac{x}{t}\right)\cos \frac{x}{t} + C'\left(t\right) = \exp\left(2\sin \frac{x}{t}\right)x\cos \frac{x}{t}-t\exp\left(2\sin \frac{x}{t}\right)  .\]
Así, tenemos que $\displaystyle C'\left(t\right)= 0 $, por lo que $\displaystyle C\left(t\right) $ es una constante y tendremos que la solución de la EDO será
\[-\frac{t ^{2}}{2}\exp\left(2\sin \frac{x}{t}\right) = C, \; C \in \R .\]
\end{eg}
\begin{eg}[Hoja 2, Ejercicio 11 apartado (a)]
Busca un factor integrante de tipo $\displaystyle \mu\left(x\right) $ o $\displaystyle \mu\left(y\right) $ y resuelve la EDO. Consideremos la EDO $\displaystyle \left(x + y\right)dx + dy = 0 $. En este caso tenemos que $\displaystyle M\left(x,y\right)=x + y $ y $\displaystyle N\left(x,y\right)= 1 $. Tenemos que 
	\[\frac{\partial M}{\partial y}= 1, \quad \frac{\partial N}{\partial x}= 0 .\]
	Como $\displaystyle \frac{\partial M}{\partial y}\neq \frac{\partial N}{\partial x} $, la EDO no es exacta, por lo que debemos multiplicar por un factor integrante. Cogemos $\displaystyle \mu\left(x\right) $, tendremos que $\displaystyle \widetilde{M}\left(x,y\right)=\mu\left(x\right)\left(x+y\right) $ y $\displaystyle \tilde{N}\left(x,y\right)=\mu\left(x\right) $, así 
	\[\frac{\partial \widetilde{M}}{\partial y} = \mu\left(x\right) \quad \text{y} \quad \frac{\partial \tilde{N}}{\partial x}=\mu'\left(x\right) .\]
Así, buscamos que $\displaystyle \mu\left(x\right)=\mu'\left(x\right) $, por lo que cogemos $\displaystyle \mu\left(x\right)= e^{x} $. Así, podemos resolver la nueva EDO como una EDO exacta. Buscamos $\displaystyle F : \R ^{2} \to \R $ tal que 
\[\frac{\partial F}{\partial x}= \widetilde{M}\left(x,y\right)=e^{x}\left(x+y\right) \quad \text{y} \quad \frac{\partial F}{\partial y}=\tilde{N}\left(x,y\right)=e^{x}.\]
Así, tendremos que 
\[F\left(x,y\right)=\int e^{x} \; dy = e^{x}y + C\left(x\right) .\]
Buscamos $\displaystyle C\left(x\right) $:
\[e^{x}y + C'\left(x\right)=e^{x}\left(x+y\right) \Rightarrow C'\left(x\right)=e^{x}x .\]
Por tanto, tendremos que $\displaystyle C\left(x\right)=e^{x}x-e^{x} $ y así nos queda que la solución de la EDO será
\[e^{x}y + e^{x}x - e^{x} = C, \; C \in \R .\]
\end{eg}
\section{Cambios de variable}
\begin{definition}[EDO de Bernoulli]
Se denomina \textbf{EDO de Bernoulli} a las EDO's que tienen la forma 
\[x'=p\left(t\right)x + q\left(t\right)x^{\alpha }, \; \alpha \in \R .\]
\end{definition}
\begin{observation}
Podemos ver que 
\begin{itemize}
\item Si $\displaystyle \alpha = 0 $, es una EDO lineal no homogénea.
\item Si $\displaystyle \alpha = 1 $, es una EDO lineal homogénea. 
\end{itemize}
\end{observation}
\begin{observation}[Método de resolución]
	Supongamos que $\displaystyle \alpha \not\in \left\{ 0,1\right\}  $. Podemos escribir
	\[x^{-\alpha }x' = p\left(t\right)x^{1-\alpha }+q\left(t\right) .\]
Sea $\displaystyle z\left(t\right)=x^{1-\alpha } $, entonces $\displaystyle z'\left(t\right)=\left(1-\alpha \right)x^{-\alpha }x' $. Así, la EDO nos queda
\[\frac{z'}{1-\alpha }=p\left(t\right)z + q\left(t\right) ,\]
que es una EDO lineal no homogénea. 
\end{observation}
\begin{eg}[Hoja 2, Ejercicio 12 apartado (d)]
Resolver la EDO
\[3xy^{2}\frac{dy}{dx}+y^{3}=\frac{8}{x} .\]
Podemos reescribir la EDO de forma que 
\[y' = -\frac{y}{3x}+\frac{8}{3x^{2}}y^{-2} .\]
Se trata de una EDO de Bernoulli de grado $\displaystyle \alpha = -2 $. Podemos escribir
\[y^{2}y' = \frac{-1}{3x}y^{3}+\frac{8}{3x^{2}} .\]
Cogiendo $\displaystyle z\left(x\right)=y^{3} $, tenemos que $\displaystyle 3y^{2}y' = z' $, así nos queda
\[\left(E\right) \quad z' = -\frac{z}{x}+\frac{8}{x^{2}} .\]
Se trata de una EDO lineal. La EDO homogénea asociada es $\displaystyle \left(E_{h}\right) z_{h}' = -\frac{z_{h}}{x} $, que podemos calcular por separación de variables
\[\frac{z_{h}'}{z_{h}}= -\frac{1}{x} \iff z_{h}=\frac{C}{x}, \; C \in \R .\]
Calculamos $\displaystyle z_{p}\left(x\right) $ por el método de separación de constantes: $\displaystyle z_{p}\left(x\right)=\frac{C\left(x\right)}{z} $, entonces $\displaystyle z_{p}' = \frac{C'}{x}-\frac{C}{x^{2}} $. Queremos que 
\[z'_{p}=-\frac{1}{x}z_{p} + \frac{8}{x^{2}} \Rightarrow C'\left(x\right)=\frac{8}{x} \Rightarrow C\left(x\right)=8\ln x .\]
Así, tenemos que la solución general de $\displaystyle \left(E\right) $ es
\[z\left(x\right)= \frac{C}{x} + \frac{8\ln x}{x} .\]
Por tanto, la solución general de nuestra EDO es
\[y^{3}=\frac{C}{x}+\frac{8\ln x}{x} .\]
\end{eg}

\begin{definition}[EDO de Riccati]
Se llama \textbf{EDO de Riccati} a las EDO's que tiene la forma
\[x' = r\left(t\right)+p\left(t\right)x+q\left(t\right)x^{2} .\]
\end{definition}
\begin{observation}
Podemos obtener la solución general a partir de una solución particular conocida $\displaystyle x_{0}\left(t\right) $. El cambio de variable $\displaystyle z\left(t\right)= x\left(t\right)-x_{0}\left(t\right) $ convierte a la EDO de Riccati en una de Bernoulli. En efecto, tendremos que $\displaystyle x' = z' + x_{0}' $ y como $\displaystyle x_{0} $ es solución tendremos que 
\[x_{0}' = r\left(t\right) + p\left(t\right)x_{0}+q\left(t\right)x_{0} .\]
Sustituyendo en la EDO,
\[z' + x_{0}' = r\left(t\right) + p\left(t\right)z + p\left(t\right)x_{0}+ q\left(t\right)z^{2} + q\left(t\right)x_{0}^{2} +2q\left(t\right)zx_{0} .\]
\[ \Rightarrow z' = \left(p\left(t\right)+2q\left(t\right)x_{0}\right)z + q\left(t\right)z^{2} .\]
\end{observation}
\begin{eg}[Hoja 2, Ejercicio 13]
Sabiendo que $\displaystyle x\left(t\right)=\tan t $ es una solución particular de
\[\frac{dx}{dt}\sin\left(2t\right)x - x^{2} = 1 - 2\sin ^{2}t .\]
Hallemos la solución general. Como es una EDO de Riccati, hacemos el cambio de variable $\displaystyle x\left(t\right)=z\left(t\right)+ \tan t $ para convertirla en una EDO de Bernoulli. Tenemos que
\[x'\left(t\right)=z'\left(t\right)+\frac{1}{\cos ^{2}t} .\]
Así, tenemos que 
\[x' = z' + \frac{1}{\cos ^{2}t} = \sin\left(2t\right)z + \sin2t \tan t + z^{2} + \tan ^{2}t + 2\tan t z+1-2\sin ^{2}t .\]
\[\Rightarrow z' = \left(\sin 2t + 2\tan t\right)z + z^{2} .\]
Para resolver esto aplicamos el método anterior para resolver las EDO's de Bernoulli. 
\end{eg}
\begin{eg}[Hoja 2, Ejercicio 16 apartado (b)]
Nos preguntan si la EDO de Riccati $\displaystyle y' + y + y^{2} + e^{x} = 0 $ se transforma en una EDO lineal de orden 2, mediante el cambio $\displaystyle y = \frac{z'}{z} $. Tendremos que 
\[y' = \frac{z'' z-\left(z'\right)^{2}}{z^{2}} .\]
Así, nos queda
\[\frac{z''}{z}-\frac{z'^{2}}{z^{2}}+\frac{z'}{z}+\frac{z'^{2}}{z^{2}}+e^{x} =0 \iff z'' + z' + e^{x}z = 0 .\]
Claramente se trata de una EDO homogénea de orden 2, por lo que el enunciado es verdadero. 
\end{eg}
\section{EDOS de segundo orden reducibles}
En su forma general una EDO de orden 2 es de la forma 
\[F\left(t, x, x', x''\right)=0 .\]
Si faltan $\displaystyle x $ o $\displaystyle t $ en esa expresión, mediante un cambio de variable la EDO se convierte en una de orden 1. Podemos distinguir varios casos:

\begin{itemize}
\item \textbf{Primer caso.} Consideremos el caso $\displaystyle F\left(x'', x', t\right)= 0 $. Tomamos $\displaystyle x'\left(t\right)=z\left(t\right) $ y $\displaystyle x''\left(t\right)=z'\left(t\right) $, así la EDO nos queda $\displaystyle F\left(z', z, t\right)= 0 $. 
\item \textbf{Segundo caso.} Consideremos $\displaystyle F\left(x'', x', x\right)= 0 $. Hacemos el cambio $\displaystyle x'\left(t\right)=z\left(t\right) $ y tenemos que 
	\[x''\left(t\right)=\frac{d}{dt}x' = \frac{d}{dt}\frac{dx}{dx}x' = \frac{d}{dx}\left(\frac{dx}{dt}\right)x'=z\frac{d}{dx}z .\]
Así, la EDO nos queda $\displaystyle F\left(\frac{d}{dx}z, z, x\right)= 0 $.
\end{itemize}

\begin{eg}[Hoja 2, Ejercicio 10 apartado 10]
	Consideremos la EDO
	\[y'' + \left(y'\right)^{2}= 1 .\]
Estamos en el caso número 1. Así, podemos coger $\displaystyle z\left(x\right)=y'\left(x\right) $ y obtenemos que la EDO se convierte en $\displaystyle z' + z^{2}=1 $, que es una EDO por separación de variables:
\[\frac{z'}{1-z^{2}}=1 \iff \frac{1}{2}\ln \left(\frac{1+z}{1-z}\right)=x+C \iff \frac{1+z}{1-z}=Ce^{2x}, \; C \in \R .\]
Despejamos $\displaystyle z = y' $ 
\[y' =z = \frac{Ce^{2x}-1}{Ce^{2x}+1}, \quad C \in \R .\]
Entonces, tendremos que 
\[y = \int \frac{Ce^{2x}-1}{Ce^{2x}+1} \; dx = \frac{1}{2}\ln C_{1}e^{2x} - \ln C_{1}e^{2x} + \ln \left|C_{1}e^{2x}+1\right|+C_{2}, \quad C_{1}, C_{2} \in \R .\]
\end{eg}
\begin{eg}[Hoja 2, Ejercicio 12 apartado 7]
Consideremos el problema de valor inicial
\[
\begin{cases}
y'' +y=y'^{2} \\
y\left(1\right)=\frac{1}{4} \\
y'\left(1\right)=\frac{1}{2}
\end{cases}
.\]
Cogemos $\displaystyle z\left(x\right)=y' $. Así, tendremos que 
\[y'' = \frac{d}{dx}y' = \frac{dy}{dx}\frac{d}{dx}y' = z\frac{d}{dy}z .\]
La EDO nos queda
\[z\frac{dz}{dy} +y=z^{2} \iff \frac{dz}{dy} = z - \frac{y}{z},\]
que es una Bernoulli de grado $\displaystyle \alpha = -1 $. Se puede comprobar que también es de factor integrante $\displaystyle \mu\left(y\right) $.  En efecto, podemos reescribir la EDO de la forma
\[z' = z-yz^{-1} .\]
Hacemos el cambio $\displaystyle u\left(y\right)=z^{2} $ de forma que $\displaystyle u'\left(y\right)=2zz' $, por lo que la EDO nos queda de la forma
\[\frac{u'}{2}+y=u ,\]
que es una EDO lineal no homogénea. Si la resolvemos obtenemos
\[u = Ce^{2y}+y+\frac{1}{2} \Rightarrow z^{2} = Ce^{2y}+y+\frac{1}{2} .\]
Calculamos el valor de $\displaystyle C $. En $\displaystyle x = 1 $, tenemos que $\displaystyle y = -\frac{1}{4} $ por lo que $\displaystyle z = \frac{1}{2} $ y se tiene que 
\[\frac{1}{4}=Ce^{-\frac{1}{2}}+\frac{1}{4}\Rightarrow C = 0 .\]
Por tanto, 
\[z = \sqrt{y + \frac{1}{2}} ,\]
las condiciones iniciales nos obligan a quedarnos con la rama positiva. Así, nos queda
\[y' = \sqrt{y + \frac{1}{2}} ,\]
que es una EDO de variables separadas, y obtenemos
\[2\sqrt{y + \frac{1}{2}} = x + C \Rightarrow y + \frac{1}{2} = \frac{\left(x+C\right)^{2}}{4} .\]
Si sustituimos las condiciones iniciales tenemos que 
\[\frac{1}{4} = \frac{\left(1 + C\right)^{2}}{4} \Rightarrow C = 0 .\]
Así, la solución de nuestra EDO es
\[y = \frac{x^{2}}{4}-\frac{1}{2} .\]
\end{eg}
\section{Ecuación de Clairaut y de Legendre}
\begin{definition}[Ecuaciones de Clairaut y Legendre]
Se denomina \textbf{ecuación de Clairaut} a las EDO's de la forma
\[x = tx' + f\left(x'\right) ,\]
donde $\displaystyle f $ es derivable.  
\end{definition}
\begin{observation}[Método de resolución de las ecuaciones de Clairaut]
Busco una solución que sea de clase $\displaystyle \mathcal{C}^{2} $. En este caso, tenemos que 
\[x' = x' + tx'' + f'\left(x'\right)x'' \Rightarrow x''\left(t+f'\left(x'\right)\right)=0 .\]
Tenemos dos posibles soluciones:
\begin{itemize}
\item La solución $\displaystyle x'' = 0 $ define una familia de soluciones.
\item La solución $\displaystyle t + f'\left(x'\right) = 0 $ va a definir la envolvente de dicha familia que es solución singular de la EDO. 
\end{itemize}
En el primer caso, como $\displaystyle x'' = 0 $ debe ser que $\displaystyle x' = C $ y si volvemos a la EDO original tenemos que 
\[x_{C}\left(t\right)=tC + f\left(C\right), \; C \in \R ,\]
es una familia de soluciones de la EDO. En el otro caso, tenemos que $\displaystyle t +f'\left(x'\right)=0 $. Como $\displaystyle \left(t, x\left(t\right)\right) $ dependen de $\displaystyle x' $, se suele dar la solución en forma paramétrica $\displaystyle x'\left(t\right)=p\left(t\right) $, mi solución es
\[
\begin{cases}
t = f'\left(p\right) \\
x = f'\left(p\right)p + f\left(p\right)
\end{cases}
.\]
\end{observation}
\begin{eg}[Hoja 2, Ejercicio 15 apartado (b)]
Consideremos la EDO 
\[ y = xy' + \frac{a}{2y'}, \; a \in \R, \; a \neq 0 .\]
Derivando obtenemos 
\[y' = y' + xy'' -\frac{a}{4}\left(y'\right)^{2}y'' \Rightarrow y''\left(x-\frac{a}{4}\left(y'\right)^{-2}\right)=0 .\]
Obtenemos las siguientes soluciones:
\begin{itemize}
\item La familia monoparamétrica $\displaystyle y_{C}\left(x\right)=Cx + \frac{a}{2^{C}} $. 
\item Tendremos que 
	\[
	\begin{cases}
	x = -\frac{a}{4}p^{-2} \\
	y = - \frac{a}{4}p^{-1}+\frac{1}{2p}
	\end{cases}
	.\]
\end{itemize}
\end{eg}
\begin{eg}
Consideremos la EDO $\displaystyle x = tx' + f\left(x'\right) $ con $\displaystyle f $ derivable. Tenemos que 
\[x' = x' + tx'' + 2x'x'' \iff x''\left(t+2x'\right)=0 \iff 
\begin{cases}
x'' = 0 \Rightarrow x_{C}= tC + C^{2} \\
t = -2x'
\end{cases}
.\]
En el segundo caso parametrizamos $\displaystyle x' = p $ y nos queda
\[ 
\begin{cases}
t = -2p \\
x = -2pp +p^{2}=-\frac{t ^{2}}{2}+\frac{t ^{2}}{4}=-\frac{t ^{2}}{4}
\end{cases}
.\]
Así, obtenemos la solución singular $\displaystyle x\left(t\right)=-\frac{t ^{2}}{4} $. 
\end{eg}
\begin{definition}[Ecuación de Legendre]
Las \textbf{ecuaciones de Legendre} son las EDO's de la forma
\[x=tg\left(x'\right) + f\left(x'\right) ,\]
con $\displaystyle f $ y $\displaystyle g $ derivables \footnote{Si hacemos $\displaystyle g\left(x\right)=x $, obtenemos la de Clairaut.}.
\end{definition}

\begin{observation}[Método de resolución de las ecuaciones de Legendre]
Consideremos la EDO $\displaystyle x = tg\left(x'\right)+f\left(x'\right) $, con $\displaystyle f $ y $\displaystyle g $ derivables. Vamos a buscar una solución de clase $\displaystyle \mathcal{C}^{2} $, de esta forma
\[x' = g\left(x'\right)+tg'\left(x'\right)x'' + f'\left(x'\right)x'' .\]
Parametrizamos $\displaystyle x' = p\left(t\right) $, así
\[p -g\left(p\right)=\frac{dp}{dt}\left(tg'\left(p\right)+f\left(p\right)\right) .\]
Distinguimos dos casos:
\begin{itemize}
\item $\displaystyle tg'\left(p\right)+f\left(p\right)\neq 0 $, entonces
	\[\frac{p-g\left(p\right)}{tg'\left(p\right)+g'\left(p\right)} = \frac{dp}{dt} \iff \frac{dt}{dp} = \frac{tg'\left(p\right)+f'\left(p\right)}{p - g\left(p\right)} = \frac{g'\left(p\right)}{p - g\left(p\right)}t + \frac{f'\left(p\right)}{p - g\left(p\right)} .\]
	Se trata de una EDO lineal de orden 1. 
\item $\displaystyle tg'\left(p\right)+f\left(p\right)=0 $ produce la solución singular. 
\end{itemize}
\end{observation}
